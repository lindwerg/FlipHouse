```python
from __future__ import annotations
import math
from dataclasses import dataclass

ONE_EURO_MIN_CUTOFF: float = 0.8
ONE_EURO_BETA: float = 0.15
ONE_EURO_D_CUTOFF: float = 1.0
EPS: float = 1e-6


def _alpha(t_e: float, cutoff: float) -> float:
    r = 2.0 * math.pi * cutoff * t_e
    return r / (r + 1.0)


@dataclass
class OneEuroFilter:
    """1-D adaptive low-pass (Casiez et al.). Mutable by design ŌĆö carries filter state."""
    min_cutoff: float = ONE_EURO_MIN_CUTOFF
    beta: float = ONE_EURO_BETA
    d_cutoff: float = ONE_EURO_D_CUTOFF
    _x_prev: float | None = None
    _dx_prev: float = 0.0
    _t_prev: float | None = None

    def reset(self, x: float, t: float) -> None:
        """Snap state to (x, t) ŌĆö the scene-cut hard cut. Clears velocity."""
        self._x_prev, self._dx_prev, self._t_prev = x, 0.0, t

    def filter(self, x: float, t: float) -> float:
        if self._t_prev is None:
            self._x_prev, self._t_prev = x, t
            return x
        t_e = max(t - self._t_prev, EPS)          # EPS guard: duplicate timestamp can't /0
        dx = (x - self._x_prev) / t_e
        edx = self._dx_prev + _alpha(t_e, self.d_cutoff) * (dx - self._dx_prev)
        cutoff = self.min_cutoff + self.beta * abs(edx)
        x_hat = self._x_prev + _alpha(t_e, cutoff) * (x - self._x_prev)
        self._x_prev, self._dx_prev, self._t_prev = x_hat, edx, t
        return x_hat
```

> **`-rc_mode` DROPPED (MEDIUM reframe-edge + LOW contract-drift accepted).** `-rc_mode bitrate` is a libopenh264-private option whose spelling varies by build and is **untestable on the dev box** (no libopenh264). libopenh264 already uses ABR when `-b:v` is set, so `-b:v 6M -maxrate 8M -bufsize 12M` is the portable, version-robust form. This removes a string the unit test could "prove present" but ffmpeg might reject only at live render.
> **`-allow_skip_frames 0` ADDED (MEDIUM finding).** Prevents libopenh264 dropping frames to hit bitrate on high-motion 9:16 clips (silent stutter that passes every dim/codec check). `-maxrate 8M` (> `-b:v 6M`) gives overshoot headroom so the encoder doesn't need to skip. Documented inline.

### 4a. Speaker-crop (TRACK) ŌĆö `CropBox.mode == "CROP"`
`-vf`: `crop={box.w}:{box.h}:{box.x}:{box.y},scale=1080:1920,setsar=1`
Full command:
```
ffmpeg -hide_banner -nostdin -loglevel error \
  -ss {start} -i {src} -t {end-start} \
  -vf "crop={w}:{h}:{x}:{y},scale=1080:1920,setsar=1" \
  -c:v libopenh264 -profile high -allow_skip_frames 0 -b:v 6M -maxrate 8M -bufsize 12M -g 60 -pix_fmt yuv420p \
  -c:a aac -b:a 128k -ar 48000 -ac 2 \
  -movflags +faststart \
  {out_path}
```
`box.w/h/x/y` are Python-computed even ints (never float ffmpeg expressions). `-ss` before `-i` (fast+accurate re-encode seek, mirrors cutter). Output is a real seekable file (`+faststart` needs a seekable output).

### 4b. Blur-pad fallback ŌĆö `CropBox.mode == "BLURPAD"`
`-vf` (foreground **fit, not crop** ŌĆö LOW finding fixed: `decrease` so a tall fg is letterboxed against the blur, never head-cropped):
```
split=2[bg][fg];[bg]scale=1080:1920:force_original_aspect_ratio=increase,crop=1080:1920,gblur=sigma=20[bgb];[fg]scale=1080:1920:force_original_aspect_ratio=decrease[fgs];[bgb][fgs]overlay=(W-w)/2:(H-h)/2,setsar=1
```
Same encode tail / `-ss`-`-i`-`-t` ordering / real-file output.

### 4c. Encoder preflight (S6)
`assert_render_codecs` runs `ffmpeg -hide_banner -encoders`, asserts both `libopenh264` and `aac` present, else `logger.critical(...)` + `raise RuntimeError(f"ffmpeg missing required delivery encoders: {missing}")`. Exact mirror of `preflight.assert_clip_codecs`. **Boot-only, not auto-wired** (importing on a libx264 dev box must not raise). The same boot entrypoint that wires `assert_clip_codecs` should also wire `assert_render_codecs` (HIGH finding ŌĆö flag in PR).

> **ffprobe codec note:** libopenh264's `codec_name` is **`h264`**, not `libopenh264`. The golden asserts `probe_video_codec(...) == "h264"`.

## 5. Reframe algorithm pseudocode

```
# ---- MediapipeSpeakerRegionSelector.select_speaker_region(src, start, end, scene_cut_times) ----
src_w, src_h = _probe_dims_fn(src)                          # Sdims
frames = _sample_faces(src, start, end, sample_fps)         # S1: per-frame tuple[FaceBox,...]
cuts = [c - start for c in scene_cut_times if start <= c < end]   # PRECOMPUTED ŌĆö no rescan

samples = []
prev_cx, cooldown = None, 0
for i, faces in enumerate(frames):
    t = i / sample_fps
    chosen, cooldown = select_active_face(faces, prev_cx, cooldown, src_w)
    cx = chosen.center_x if chosen else None
    if chosen is not None: prev_cx = cx
    samples.append(RawSample(t=t, center_x=cx, face_count=len(faces)))
return build_trajectory(samples, cuts, src_w, src_h)


# ---- select_active_face(faces, prev_cx, cooldown_left, src_w)  [PURE] ----
if not faces:
    return None, max(0, cooldown_left - 1)
sticky_radius = STICKY_RADIUS_FRAC * src_w
def score(f):
    s = f.area
    if prev_cx is not None and abs(f.center_x - prev_cx) < sticky_radius:
        s *= STICKINESS_BONUS
    return s
if prev_cx is not None and cooldown_left > 0:               # cooling down ŌåÆ anti-ping-pong
    best = min(faces, key=lambda f: abs(f.center_x - prev_cx))
    return best, cooldown_left - 1
best = max(faces, key=score)
switched = prev_cx is None or abs(best.center_x - prev_cx) >= sticky_radius
return best, (SWITCH_COOLDOWN_FRAMES if switched else max(0, cooldown_left - 1))


# ---- build_trajectory(samples, cuts, src_w, src_h)  [PURE] ----
deadband = deadband_frac * src_w
euro = OneEuroFilter()
held = src_w / 2.0
keyframes, face_counts = [], []
for s in samples:
    if any(abs(s.t - cut) <= SNAP_EPS_S for cut in cuts):   # scene-cut snap
        euro.reset(s.center_x if s.center_x is not None else held, s.t)
        held = s.center_x if s.center_x is not None else held
    if s.center_x is None:
        keyframes.append(CropKeyframe(s.t, None, GENERAL_MARK)); continue
    face_counts.append(s.face_count)
    if abs(s.center_x - held) < deadband:
        cx = held
    else:
        cx = euro.filter(s.center_x, s.t); held = cx
    keyframes.append(CropKeyframe(s.t, cx, TRACK_MARK))
avg_faces = mean(face_counts) if face_counts else 0.0
if avg_faces == 0.0 or avg_faces > general_face_max:        # group-shot / no-face override
    keyframes = [CropKeyframe(kf.t, None, GENERAL_MARK) for kf in keyframes]
return CropTrajectory(tuple(keyframes), src_w, src_h)


# ---- render.py per-clip MVP consumption (single static crop) ----
span = end - start
if span <= 0:  raise ValueError(...)                        # fail-closed
if span > MAX_CLIP_DURATION_S:  raise ClipDurationError(...) # fail-closed (HIGH 180s finding)
traj = selector.select_speaker_region(src, start, end, clip_cuts)
# dominant_center / is_general are ONE source of truth (is_general == dominant_center() is None)
if traj.is_general():
    box = CropBox(0, 0, traj.source_width, traj.source_height, BLURPAD_MODE)
else:
    box = compute_crop_box(traj.source_width, traj.source_height, traj.dominant_center())
    # compute_crop_box may itself downgrade to BLURPAD if source too narrow
_render_fn(src, start, end, box, out, target_w, target_h, bitrate)   # ONE ffmpeg call
w, h = _probe_fn(out)                                        # S3
if (w, h) != (target_w, target_h):  raise DimensionMismatchError(...)
```

> **GRAFT (A) ŌĆö YAGNI boundary:** MVP renders **ONE static crop window per clip** (`dominant_center` = median of TRACK centers) = a single mockable subprocess. The full per-keyframe One-Euro/scene-cut-snapped `CropTrajectory` is computed and **carried** for P3's per-frame `sendcmd` pan **without re-detection or refactor**.
